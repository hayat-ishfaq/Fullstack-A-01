Usage Examples

HTML

<!-- Favicon -->
<link rel="icon" type="image/svg+xml" href="assets/favicon.svg">

<!-- Logo Image -->
<img src="assets/logo-full.svg" alt="HOTSPRING" class="w-20 h-20">

<!-- Loading Spinner -->
<img src="assets/loading.svg" alt="Loading..." class="w-8 h-8">

<!-- Success Icon -->
<img src="assets/success.svg" alt="Success" class="w-16 h-16">

CSS

/* Background Pattern */
body {
    background-image: url('assets/background-pattern.svg');
    background-repeat: repeat;
    background-size: 100px 100px;
}

JavaScript/jQuery

// Show loading spinner
$('#loading').html('<img src="assets/loading.svg" alt="Loading..." class="inline-block w-6 h-6">');

// Show success message
$('#status').html('<img src="assets/success.svg" alt="Success" class="w-12 h-12">');
